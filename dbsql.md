
CREATE TABLE person (
id BIGINT AUTO_INCREMENT PRIMARY KEY,
building_number VARCHAR(20) NOT NULL COMMENT '楼号',
unit_number INT comment '单元号',
room_number VARCHAR(100) not null COMMENT '房号',
name VARCHAR(50) COMMENT '姓名',
id_card VARCHAR(20) COMMENT '身份证号',
age INT COMMENT '年龄',
gender TINYINT COMMENT '性别：0未知，1男，2女',
is_permanent tinyint COMMENT '是否常住：0未知，1是，2否',
housing_situation VARCHAR(100) COMMENT '住房情况',
property_nature VARCHAR(200) COMMENT '房屋性质',
registered_residence_type tinyint COMMENT '户籍情况:0.未知1.东湖户籍2.北京市其他户籍3.外地户籍',
registered_residence VARCHAR(200) COMMENT '户籍地',
telephone VARCHAR(50) COMMENT '联系方式',
first_contact VARCHAR(50) COMMENT '第一联系人',
elder_relationship VARCHAR(50) COMMENT '关系(老人之)',
elder_contact_phone VARCHAR(20) COMMENT '联系电话',
special_situation TEXT COMMENT '特殊情况',
has_electric_car ENUM('是', '否') COMMENT '电动车',
disability_level VARCHAR(100) COMMENT '失能等级',
is_low_income tinyint COMMENT '是否低保0否，1是',
is_low_income2 tinyint COMMENT '是否低收入0否，1是',
is_destitute tinyint COMMENT '是否特困供养0否，1是',
is_family_planning_special tinyint COMMENT '是否计划生育特殊家庭0否，1是',
disability_category VARCHAR(100) COMMENT '残疾类别及等级',
is_living_alone tinyint COMMENT '是否独居0否1是',
is_empty_nest tinyint COMMENT '是否空巢0否1是',
is_orphaned tinyint COMMENT '是否孤寡0否1是',
is_needs_focus tinyint COMMENT '是否需重点关注0否1是',
other_situation TEXT COMMENT '其它情况',
-- 电动车信息
license_plate VARCHAR(20) COMMENT '车牌号码',
brand_model VARCHAR(100) COMMENT '品牌/型号',
is_in_group tinyint COMMENT '是否入群0否1是',
is_private_message tinyint COMMENT '是否私信0否1是',
has_pet tinyint COMMENT '家有宠物0否1是',
last_contact_time VARCHAR(50) COMMENT '末次联系时间',
other_info TEXT COMMENT '其他',
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
INDEX idx_building (building_number),
INDEX idx_address (unit_number),
INDEX idx_name (name),
INDEX idx_id_card (id_card),
INDEX idx_age (age)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='人员信息台账';
